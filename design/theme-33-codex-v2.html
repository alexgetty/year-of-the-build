// Golden ratio type scale generator
const goldenScale = (base, steps) => {
  const phi = 1.618;
  return Array.from({ length: steps }, (_, i) => ({
    size: Math.round(base * Math.pow(phi, i - 2)),
    multiplier: `phi^${i - 2}`
  }));
};

// Usage
const scale = goldenScale(16, 8);
// [10, 13, 16, 26, 42, 68, 110, 178]
